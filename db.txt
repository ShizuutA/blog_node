DBname = Blog_Romain

table1 = Users {
    username varchar(255) PRIMARY KEY NOT NULL,
    password (sha1) : varchar(255) NOT NULL,
    rights ENUM('Member', 'Admin') DEFAULT 'Member'
}

table2 = Posts {
    ID int PRIMARY KEY AUTO_INCREMENT NOT NULL,
    content varchar(255) NOT NULL,
    imgname varchar(255),
    imgdata longblob,
    date DATE
}
